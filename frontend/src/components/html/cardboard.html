CSS 实现代码 金色卡牌边框样式 
.card {
  width: 350px;
  min-height: 450px;
  background-color: white;
  position: relative;
  box-shadow: 0 10px 30px rgba(0, 0, 0, 0.08);
}

/* 基础金色边框 */
.gold-border {
  border: 1px solid #d4af37;
}

/* 高级金色边框 - 使用伪元素创建双层效果 */
.fancy-border {
  border: none;
  position: relative;
  padding: 2px;
}

.fancy-border::before {
  content: '';
  position: absolute;
  top: 0;
  left: 0;
  right: 0;
  bottom: 0;
  background: linear-gradient(45deg, #d4af37, #f7ef8a, #d4af37);
  z-index: 1;
  border-radius: 4px;
}

/* 角部装饰 */
.corner-decoration {
  position: absolute;
  width: 40px;
  height: 40px;
  z-index: 3;
}

.corner-tl {
  top: 0;
  left: 0;
  border-top: 2px solid #d4af37;
  border-left: 2px solid #d4af37;
}

/* 角部花纹装饰 */
.corner-tl::before {
  content: '';
  position: absolute;
  width: 20px;
  height: 20px;
  top: -1px;
  left: -1px;
  border-top: 2px solid #d4af37;
  border-left: 2px solid #d4af37;
}

/* 花纹装饰符号 */
.ornament {
  position: absolute;
  color: #d4af37;
  font-size: 16px;
}
